    selectedRowKeys = [],

    rowKey = "id",

    const key = getRowKey(record, index);

                    type="checkbox"

                    onChange={(e) => handleSelect(record, rowIndex, e)}

                  : undefined;

                  <td key={column.key || colIndex}>
                    {column.render
                      ? column.render(cellValue, record, rowIndex)
                      : cellValue}
                  </td>
                );
              })}
            </tr>
          );
        })}
      </tbody>
    );
  };

    <div
      className={`table-container size-${size} ${bordered ? "bordered" : ""}`}
